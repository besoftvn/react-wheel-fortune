```bash
npm install --save react-wheel-fortune
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'react-wheel-fortune'
import 'react-wheel-fortune/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
